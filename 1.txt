procedure Form1.toolStripMenuItem13_Click(sender: Object; e: EventArgs);
    var d, i, k, j: integer;
sl: array[1..10] of string;
begin
    s:=TextBox1.Text;
    d:=length(s);
    k:=1;
    s1[k]:='';
    i:=1;
    while i<=d do
        if s[i] <> ' ' then
        begin
            s1[k]:=s1[k]+s[i];
            i:=i+1;
        end
        else
            begin
            k:=k+1;
            s1[k]:='';
            while (s[i] = ' ') and (i<=d) do
                i:=i+1;
            end;
            for i:=1 to k do
                TextBox1.Text:=TextBox1.Text+#13#10+sl[i];
end;

procedure Form1.pictureBox1_MouseMove(sender: Object; e: MouseEventArgs);
begin
    if (m_d) and (CheckBox1.Checked) then
    begin
        gr.DrawLine(p_c, x1, y1, e.X, e.Y);
    end;
    if (m_d) and (CheckBox2.Checked) then
        begin
        gr.DrawRectangle(p_c, x1, y1, x2, y2 );
        gr.FillRectangle(s_b, x1, y1, x2, y2 );
        end;
    if (m_d) and (CheckBox3.Checked) then
        begin
        gr.DrawEllipse(p_c, x1, y1, x2, y2);
        gr.FillEllipse(s_b, x1, y1, x2, y2);
        end;
    x1:=e.X; y1:=e.Y;
end;
